# Disable IPV6:

nano /etc/default/grub
* Add in File:
GRUB_CMDLINE_LINUX_DEFAULT="ipv6.disable=1"
GRUB_CMDLINE_LINUX="ipv6.disable=1"

update-grub

sysctl -w net.ipv6.conf.all.disable_ipv6=1
sysctl -w net.ipv6.conf.default.disable_ipv6=1
sysctl -w net.ipv6.conf.lo.disable_ipv6=1

nano /etc/sysctl.conf 
* Add in End File:
net.ipv6.conf.all.disable_ipv6=1
net.ipv6.conf.default.disable_ipv6=1
net.ipv6.conf.lo.disable_ipv6 = 1

cat /proc/sys/net/ipv6/conf/all/disable_ipv6

reboot

# Check and Alter Status on Firewall:

sudo ufw status

sudo ufw enable

sudo ufw disable

sudo ufw status verbose

--------------------------------------------------------------------------------------------------------
# Verify Rules the Firewall:

sudo ufw status numbered

--------------------------------------------------------------------------------------------------------
# Open Port on Firewall:

* All Ip:
sudo ufw allow 22/tcp

* Ip Rout Begin and End:
sudo ufw allow proto tcp from 168.197.159.173 to 177.106.78.129 port 22

* All Ports to Ip:
sudo ufw allow from 168.197.159.173

* Port to Ip:
sudo ufw allow proto tcp from 168.197.159.173 to any port 22

--------------------------------------------------------------------------------------------------------
# Delete Rule the Firewall:

sudo ufw status numbered

sudo ufw delete 1
